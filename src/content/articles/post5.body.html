This post shows a minimal example: ESP32 + A76xx LTE module connecting to an MQTT broker over MQTTS (port 8883), subscribing to a topic, publishing "Hello World", and printing received messages. We use the A76XX library (e.g. from GitHub) to talk to the module over UART.

What you need

Hardware: ESP32 (or other MCU), A76xx-based module (e.g. A7680, A7670), SIM with data, antenna. Connect the module UART to the MCU (e.g. TX→16, RX→17 on ESP32).

Library: A76XX (provides A76XX, A76XXMQTTClient, A76XXSecureClient). Optional: StreamDebugger to echo AT traffic to Serial, ArduinoJson if you later extend to JSON payloads.

For MQTTS the module must have the broker’s CA certificate stored. Run the SaveCertificate sketch once to write the CA (e.g. PEM) to the module; then the main sketch uses that cert by name.

Step 1: Save CA certificate to the module

Run this sketch once. It sends the broker’s CA certificate (PEM) to the module and saves it under a name (e.g. mosquitto.pem). Use the same name in the main sketch with mqtt.setCaCert(...).

#include <StreamDebugger.h>
#include "A76XX.h"

#define DEBUG_AT true
#if DEBUG_AT
  StreamDebugger SerialAT(Serial1, Serial);
#else
  #define SerialAT Serial1
#endif

static const char* mosquittoCert PROGMEM = R"EOF(
-----BEGIN CERTIFICATE-----
... paste your broker CA PEM here ...
-----END CERTIFICATE-----)EOF";

A76XX modem(SerialAT);
A76XXSecureClient client(modem);

#define PIN_TX 17
#define PIN_RX 16

void setup() {
  Serial.begin(115200);
  Serial1.begin(115200, SERIAL_8N1, PIN_RX, PIN_TX);
  delay(3000);

  if (!modem.init()) { Serial.println("modem error"); while(1) {} }
  if (!client.writeCaCert(mosquittoCert, "mosquitto.pem")) {
    Serial.println("write cert error"); while(1) {}
  }
  Serial.println("Certificate saved.");
  modem.powerOff();
}

void loop() {}

Step 2: Main sketch – send and receive (Hello World)

After the certificate is stored, use this sketch. It initializes the modem, waits for network registration, connects GPRS (APN), sets the CA cert by name, starts the MQTT client, connects to the broker with optional LWT, subscribes to /test/hello, and in loop calls modem.listen(), reads any incoming MQTT messages and prints them, and publishes "Hello World" every 5 seconds.

#include <StreamDebugger.h>
#include "A76XX.h"

#define DEBUG_AT true
#if DEBUG_AT
  StreamDebugger SerialAT(Serial1, Serial);
#else
  #define SerialAT Serial1
#endif

const char* deviceId   = "a76xx_hello_01";
const char* server     = "your-broker.com";
const int   port       = 8883;
const char* apn        = "your-apn";
const char* will_topic   = "/data/a76xx_hello_01";
const char* will_message = "{\"isAlive\":0}";
#define MOSQUITTO_CA_CERTNAME "mosquitto.pem"

#define PIN_TX 17
#define PIN_RX 16

A76XX modem(SerialAT);
A76XXMQTTClient mqtt(modem, deviceId, true);

void reconnect() {
  if (!mqtt.connect(server, port, true, 60, NULL, NULL, will_topic, will_message, 0)) {
    Serial.println("MQTT connect failed");
    return;
  }
  mqtt.subscribe("/test/hello");
  Serial.println("Subscribed /test/hello");
}

void setup() {
  Serial.begin(115200);
  Serial1.begin(115200, SERIAL_8N1, PIN_RX, PIN_TX);
  delay(3000);

  modem.reset();
  delay(1000);
  if (!modem.init()) { Serial.println("modem error"); while(1) {} }
  if (!modem.waitForRegistration()) { Serial.println("registration timeout"); while(1) {} }
  if (!modem.GPRSConnect(apn)) { Serial.println("GPRS failed"); while(1) {} }
  if (!mqtt.setCaCert(MOSQUITTO_CA_CERTNAME)) { Serial.println("setCaCert error"); while(1) {} }
  if (!mqtt.begin()) { Serial.println("mqtt.begin error"); while(1) {} }

  if (!mqtt.connect(server, port, true, 60, NULL, NULL, will_topic, will_message, 0)) {
    Serial.println("connect error"); while(1) {}
  }
  mqtt.subscribe("/test/hello");
  Serial.println("Ready.");
}

void loop() {
  modem.listen(100);

  while (mqtt.messageAvailable() > 0) {
    MQTTMessage_t msg = mqtt.getMessage();
    Serial.print("RX [");
    Serial.print(msg.topic);
    Serial.print("] ");
    Serial.println(msg.payload);
  }

  static unsigned long last = 0;
  if (millis() - last >= 5000) {
    last = millis();
    mqtt.publish("/test/hello", "Hello World", 0, 60, false);
    Serial.println("Published: Hello World");
  }
}

Summary

Run the SaveCertificate sketch once to store the broker CA on the module. Then run the main sketch: it connects over MQTTS, subscribes to /test/hello, publishes "Hello World" every 5 seconds, and prints any message received on that topic. Replace server and apn with your broker host and cellular APN.
