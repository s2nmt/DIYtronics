    MQTTS on W5500 and ESP32

    This post shows a minimal example: ESP32 + W5500 Ethernet connecting to an MQTT broker over TLS (MQTTS, port 8883), subscribing to a topic, publishing a simple "Hello World" message, and printing any received messages. Useful when Wi‑Fi is not available or you want a stable wired link for secure MQTT.

    What you need

    Hardware: ESP32, W5500 Ethernet module, Ethernet cable. Wire W5500 to ESP32 SPI (e.g. CS = GPIO 5) and 3.3V/GND.

    Libraries: SPI, Ethernet (W5500), SSLClient, PubSubClient. You also need a certificates.h file that defines your broker’s Root CA (e.g. ISRG Root X1 for Let’s Encrypt) in the format expected by SSLClient (e.g. BearSSL br_x509_trust_anchor). You can generate it with tools like pycert_bearssl.

    Full example: send and receive data (Hello World)

    The sketch below: initializes Ethernet (DHCP), wraps it with SSLClient using your CA, connects to the broker on port 8883, subscribes to /test/hello, and in loop publishes Hello World every 5 seconds. Incoming messages are printed in the callback.

    Main sketch

    #include <SPI.h>
#include <Ethernet.h>
#include <SSLClient.h>
#include "certificates.h"
#include <PubSubClient.h>

#define W5500_CS 5
byte mac[] = { 0xDE, 0xAD, 0xBE, 0xEF, 0xFE, 0xED };

const char* mqtt_server = "your-broker.com";
const int   mqtt_port   = 8883;
const char* clientId    = "w5500_esp32_01";

EthernetClient ethClient;
SSLClient secureClient(ethClient, TAs, (size_t)TAs_NUM, 4096, 4096);
PubSubClient mqtt(secureClient);

void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Message [");
  Serial.print(topic);
  Serial.print("]: ");
  for (unsigned int i = 0; i < length; i++) Serial.print((char)payload[i]);
  Serial.println();
}

void reconnect() {
  while (!mqtt.connected()) {
    Serial.print("Connecting MQTT...");
    if (mqtt.connect(clientId)) {
      Serial.println(" connected");
      mqtt.subscribe("/test/hello");
    } else {
      Serial.print(" failed, rc=");
      Serial.println(mqtt.state());
      delay(2000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  SPI.begin();
  Ethernet.init(W5500_CS);
  if (Ethernet.begin(mac) == 0) {
    Serial.println("Ethernet DHCP failed");
    return;
  }
  Serial.print("IP: ");
  Serial.println(Ethernet.localIP());

  mqtt.setServer(mqtt_server, mqtt_port);
  mqtt.setCallback(callback);
  mqtt.setBufferSize(512);
}

void loop() {
  if (!mqtt.connected()) reconnect();
  mqtt.loop();

  static unsigned long last = 0;
  if (millis() - last >= 5000) {
    last = millis();
    mqtt.publish("/test/hello", "Hello World", false);
    Serial.println("Published: Hello World");
  }
}

    certificates.h

    SSLClient needs one or more Root CAs in BearSSL format. Your certificates.h should define TAs_NUM and the TAs array (e.g. ISRG Root X1 for Let’s Encrypt). Example structure:

    #ifndef _CERTIFICATES_H_
#define _CERTIFICATES_H_
#ifdef __cplusplus
extern "C" { #endif

#define TAs_NUM 1
static const br_x509_trust_anchor TAs[] = {
  /* One entry per CA, e.g. ISRG Root X1 - use pycert_bearssl to generate */
};

#ifdef __cplusplus
}
#endif
#endif

    Generate the actual byte arrays with a tool like pycert_bearssl and include the resulting header in your project.

    Summary

    With the code above, the device connects to the broker over MQTTS (Ethernet + TLS), subscribes to /test/hello, publishes Hello World every 5 seconds, and prints any message it receives on that topic. Replace mqtt_server and your CA in certificates.h with your broker details.
